def native?
  [ /(u){0,1}int(\d\d_t){0,1}\s*\*{0,1}/, /float\s*\*{0,1}/, /double\s*\*{0,1}/, /char\s*\*{0,1}/, /bool/, /(std::){0,1}string/ ].each { |type|
    if @name =~ type
      return true
    end
  }
  
  false
end

def initialize(name)
  @name = name
  @valid_name = name.dup.to_s
  
  if name.include?("*")
    @is_pointer = true
    @valid_name.gsub!("*", "Ptr")
  else
  
  if not name.index(/\[\s*\]/).nil?
    @valid_name.gsub!(/\[\s*\]/, "Array")
    @is_array = true
  end
  
  end
  
  @valid_name = @valid_name.gsub("::", "_").gsub(/[&\s<>]/, "")
end
